## Usage

```rust
use entro_finance::EntroClient;

#[tokio::main]
async fn main() {
    let client = EntroClient::new("sk_test_xxx");
    
    let card = client
        .create_card("VIRTUAL", "USD")
        .await
        .unwrap();
        
    println!("Card ID: {}", card.id);
}
```
